    <script>
        const startScanBtn = document.getElementById('start-scan-btn');
        const readerDiv = document.getElementById('reader');
        const initialStateDiv = document.getElementById('initial-state');
        const manualEntryDiv = document.getElementById('manual-entry-state');
        const statusMessage = document.getElementById('status-message');
        const resultContainer = document.getElementById('result-container');
        const resultElement = document.getElementById('result');
        const resultTitle = document.getElementById('result-title');
        const showManualEntryBtn = document.getElementById('show-manual-entry');
        const backToScanBtn = document.getElementById('back-to-scan');
        const verifyIdBtn = document.getElementById('verify-id-btn');
        const manualIdInput = document.getElementById('manual-id-input');
        let html5QrcodeScanner;

        const approvedMembers = {
            "FJT123456": "Fardaan Jabir Tareen",
            "FJT987654": "Jane Doe",
            "FJT555123": "John Smith"
        };

        function showResult(isSuccess, title, message) {
            resultTitle.textContent = title;
            resultTitle.classList.toggle('text-green-400', isSuccess);
            resultTitle.classList.toggle('text-red-400', !isSuccess);
            resultElement.innerHTML = message;
            resultContainer.classList.remove('hidden');
            playBeep(isSuccess);
        }

        function onScanSuccess(decodedText, decodedResult) {
            try {
                const nameMatch = decodedText.match(/Name: (.*?),/);
                const idMatch = decodedText.match(/ID: (.*)/);

                if (nameMatch && idMatch) {
                    const name = nameMatch[1];
                    const id = idMatch[1];
                    const message = `
                        <p><strong class="font-semibold text-slate-300">Name:</strong> ${name}</p>
                        <p><strong class="font-semibold text-slate-300">Membership ID:</strong> ${id}</p>
                    `;
                    showResult(true, "Member Verified", message);
                } else {
                    showResult(false, "Scan Error", `<p class="text-red-400">Invalid FJT Services QR Code.</p>`);
                }
            } catch (e) {
                showResult(false, "Scan Error", `<p class="text-red-400">Could not read this QR Code.</p>`);
            }
        }

        function onScanFailure(error) {}

        function playBeep(isSuccess) {
            const audioContext = new (window.AudioContext || window.webkitAudioContext)();
            const oscillator = audioContext.createOscillator();
            const gainNode = audioContext.createGain();
            oscillator.connect(gainNode);
            gainNode.connect(audioContext.destination);
            oscillator.type = 'sine';
            gainNode.gain.setValueAtTime(0.5, audioContext.currentTime);
            oscillator.frequency.setValueAtTime(isSuccess ? 880 : 330, audioContext.currentTime);
            oscillator.start();
            oscillator.stop(audioContext.currentTime + 0.1);
        }

        startScanBtn.addEventListener('click', () => {
            statusMessage.textContent = 'Requesting camera access...';
            
            html5QrcodeScanner = new Html5QrcodeScanner("reader", { fps: 10, qrbox: { width: 250, height: 250 } }, false);
            
            html5QrcodeScanner.render(onScanSuccess, onScanFailure)
                .then(() => {
                    initialStateDiv.classList.add('hidden');
                    readerDiv.classList.remove('hidden');
                    readerDiv.classList.add('scanning');
                })
                .catch(err => {
                    console.error("Camera start error:", err);
                    statusMessage.innerHTML = '<strong>Camera permission denied.</strong><br>Please allow camera access in your browser settings. This site must be loaded on a secure (https) connection.';
                });
        });

        showManualEntryBtn.addEventListener('click', () => {
            initialStateDiv.classList.add('hidden');
            manualEntryDiv.classList.remove('hidden');
            resultContainer.classList.add('hidden');
        });

        backToScanBtn.addEventListener('click', () => {
            manualEntryDiv.classList.add('hidden');
            initialStateDiv.classList.remove('hidden');
            resultContainer.classList.add('hidden');
        });

        verifyIdBtn.addEventListener('click', () => {
            const memberId = manualIdInput.value.trim().toUpperCase();
            if (approvedMembers[memberId]) {
                const memberName = approvedMembers[memberId];
                const message = `
                    <p><strong class="font-semibold text-slate-300">Name:</strong> ${memberName}</p>
                    <p><strong class="font-semibold text-slate-300">Membership ID:</strong> ${memberId}</p>
                `;
                showResult(true, "Member Verified", message);
            } else {
                showResult(false, "Verification Error", `<p class="text-red-400">Invalid or unknown Membership ID.</p>`);
            }
        });
    </script>
